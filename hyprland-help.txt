自动登陆账户
sudo mkdir -p /etc/systemd/system/getty@tty1.service.d
sudo vim /etc/systemd/system/getty@tty1.service.d/autologin.conf
[Service]
ExecStart=
ExecStart=-/usr/bin/agetty --autologin xxx --noclear %I $TERM			（xxx为代指，名）

自启hyprland
shell为fish时，$ vim ~/.config/fish/config.fish
if test -z "$DISPLAY"; and test (tty) = "/dev/tty1"
    exec start-hyprland
end
starship init fish | source

shell为bash时，$ vim ~/.bash_profile
if [ -z "$DISPLAY" ] && [ "$(tty)" = "/dev/tty1" ]; then
    exec start-hyprland
fi

  		if (subject.isInGroup("wheel")) {
    			return polkit.Result.YES;
  		}
	});

查看磁盘分区$ lsblk

查看磁盘用量$ df -h
查找本地/线上软件包$ pacman -Qs/Ss <package_name>
查看软件包信息$ pacman -Qi <package_name>
查询孤立包$ sudo pacman -Qdt
删除孤立包$ sudo pacman -R $(pacman -Qdtq)
递归删除孤立软件包及其配置文件$ pacman -Qtdq | pacman -Rns -
移除该软件包、所有依赖项、配置$ sudo pacman -Rnscu <package_name>
清理不再需要的软件包数据库条目$ sudo pacman -Scc

此外，还可以使用 yay/paru（AUR助手）工具来管理 AUR（Arch User Repository）软件包
